cmake_minimum_required(VERSION 3.16)
project(AquaVisualTest)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Find Vulkan
find_package(Vulkan REQUIRED)

# Add executable
add_executable(AquaVisualTest test_aquavisual.cpp)

# Include directories
target_include_directories(AquaVisualTest PRIVATE
    ${CMAKE_CURRENT_SOURCE_DIR}
    ${Vulkan_INCLUDE_DIRS}
)

# Link libraries
target_link_libraries(AquaVisualTest PRIVATE
    ${CMAKE_CURRENT_SOURCE_DIR}/build/AquaVisual/Debug/AquaVisual.lib
    ${Vulkan_LIBRARIES}
)

# Compiler definitions
target_compile_definitions(AquaVisualTest PRIVATE
    AQUA_HAS_VULKAN=1
    VK_USE_PLATFORM_WIN32_KHR
    NOMINMAX
    WIN32_LEAN_AND_MEAN
)
