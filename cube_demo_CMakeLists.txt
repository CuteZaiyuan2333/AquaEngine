cmake_minimum_required(VERSION 3.16)
project(CubeDemo)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Add executable
add_executable(CubeDemo cube_demo.cpp)

# Include directories
target_include_directories(CubeDemo PRIVATE
    ${CMAKE_CURRENT_SOURCE_DIR}
)

# Link with the AquaVisual library
target_link_libraries(CubeDemo PRIVATE
    ${CMAKE_CURRENT_SOURCE_DIR}/build/AquaVisual/Debug/AquaVisual.lib
)

# Compiler definitions
target_compile_definitions(CubeDemo PRIVATE
    AQUA_HAS_VULKAN=1
)
